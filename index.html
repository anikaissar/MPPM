🚀 Setup Guide

1
Create four sheet tabs:
• SKU Listings — 18 cols: Timestamp | Platform SKU | Parent NH SKU | Child Codes | GST | EAN | MRP | Platform Name | Channel SKU | Shelf Life | Status | Uploader | Warehouses | Approval | Row ID | Version | Barcode | Grand Parent Code
• SKU History — 20 cols: Archived At | Platform SKU | Parent NH SKU | Child Codes | GST | EAN | MRP | Platform Name | Channel SKU | Shelf Life | Status | Uploader | Warehouses | Approval | Row ID | Version | Original Timestamp | Changed Fields | Barcode | Grand Parent Code
• Uploaders — 1 col: Name in A1, then one name per row from A2. Pre-filled with your 10 names. Add more anytime — they appear in the portal on next load.
• Bulk Upload Log — optional, for tracking bulk job history (script auto-creates entries).

const MAIN_SHEET = "SKU Listings"; const HIST_SHEET = "SKU History"; const UPLDR_SHEET = "Uploaders"; const BULKLOG_SHEET = "Bulk Upload Log"; // SKU Listings headers (18 cols) const MAIN_HEADERS = [ "Timestamp","Platform SKU","Parent NH SKU","Child Codes","GST","EAN","MRP", "Platform Name","Channel SKU","Shelf Life","Status","Uploader","Warehouses", "Approval","Row ID","Version","Barcode","Grand Parent Code" ]; // SKU History headers (20 cols) const HIST_HEADERS = [ "Archived At","Platform SKU","Parent NH SKU","Child Codes","GST","EAN","MRP", "Platform Name","Channel SKU","Shelf Life","Status","Uploader","Warehouses", "Approval","Row ID","Version","Original Timestamp","Changed Fields", "Barcode","Grand Parent Code" ]; /* ── CORS-safe response ── */ function cors(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doOptions(e) { return ContentService.createTextOutput("").setMimeType(ContentService.MimeType.TEXT); } function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const body = e.postData ? JSON.parse(e.postData.contents) : {}; const action = e.parameter.action || body.action; try { if (action === "getUploaders") return cors(getUploaders()); if (action === "check") return cors(checkSKU(e.parameter.platformSKU || body.platformSKU)); if (action === "fetch") return cors(fetchSKUs()); if (action === "fetchHistory") return cors(fetchHistory()); if (action === "submit") return cors(upsertSKU(body, getSheet(MAIN_SHEET), getSheet(HIST_SHEET))); if (action === "bulkSubmit") return cors(bulkSubmit(body)); return cors({ success: false, error: "Unknown action: " + action }); } catch(err) { return cors({ success: false, error: err.toString() }); } } /* ── get or create sheet, auto-write headers if empty ── */ function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); if (sheet.getLastRow() === 0) { const headers = name === MAIN_SHEET ? MAIN_HEADERS : name === HIST_SHEET ? HIST_HEADERS : null; if (headers) sheet.appendRow(headers); } return sheet; } /* ── uploaders ── */ function getUploaders() { const rows = getSheet(UPLDR_SHEET).getDataRange().getValues(); const names = rows.slice(1).map(r => String(r[0]).trim()).filter(Boolean); return { success: true, uploaders: names }; } /* ── check if platform SKU exists ── */ function checkSKU(platformSKU) { const rows = getSheet(MAIN_SHEET).getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]).trim() === String(platformSKU).trim()) { return { success: true, exists: true, version: rows[i][15] || 1, current: rowToObj(rows[i]) }; } } return { success: true, exists: false }; } function rowToObj(r) { return { timestamp: r[0], platformSKU: r[1], parentSKU: r[2], childCodes: r[3], gst: r[4], ean: r[5], mrp: r[6], platformName: r[7], channelSKU: r[8], shelfLife: r[9], status: r[10], uploaderName: r[11], warehouseMapping: r[12], approval: r[13], rowId: r[14], version: r[15] || 1, barcode: r[16] || "", grandParentCode: r[17] || "" }; } /* ── upsert (new or update) ── */ // Accepts optional pre-loaded sheet data for bulk mode (avoids re-reading sheet per row) function upsertSKU(data, mainSheet, histSheet, _cachedRows, _rowIndex) { const now = new Date().toISOString(); const childStr = Array.isArray(data.childCodes) ? data.childCodes.join("+") : (data.childCodes || ""); const wh = Array.isArray(data.warehouseMapping) ? data.warehouseMapping.join(", ") : (data.warehouseMapping || ""); const barcode = data.barcode || ""; const gpc = data.grandParentCode || ""; // Use cached rows in bulk mode, otherwise read from sheet const rows = _cachedRows || mainSheet.getDataRange().getValues(); const idx = _rowIndex !== undefined ? _rowIndex : (() => { for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]).trim() === String(data.platformSKU).trim()) return i; } return -1; })(); if (idx !== -1) { const old = rows[idx]; const oldVersion = Number(old[15]) || 1; const fieldMap = { "Parent NH SKU": [old[2], data.parentSKU], "Child Codes": [old[3], childStr], "GST": [old[4], data.gst], "EAN": [old[5], data.ean], "MRP": [old[6], data.mrp], "Platform Name": [old[7], data.platformName], "Channel SKU": [old[8], data.channelSKU], "Shelf Life": [old[9], data.shelfLife], "Status": [old[10], data.status], "Uploader": [old[11], data.uploaderName], "Warehouses": [old[12], wh], "Barcode": [old[16] || "", barcode], "Grand Parent Code":[old[17] || "", gpc] }; const changed = Object.entries(fieldMap) .filter(([,[o,n]]) => String(o).trim() !== String(n).trim()) .map(([k]) => k); histSheet.appendRow([ now, old[1], old[2], old[3], old[4], old[5], old[6], old[7], old[8], old[9], old[10], old[11], old[12], old[13], old[14], oldVersion, old[0], changed.join(", "), old[16] || "", old[17] || "" ]); const newRow = [ now, data.platformSKU, data.parentSKU, childStr, data.gst, data.ean, data.mrp, data.platformName, data.channelSKU, data.shelfLife, data.status, data.uploaderName, wh, "Pending", old[14], oldVersion + 1, barcode, gpc ]; mainSheet.getRange(idx + 1, 1, 1, 18).setValues([newRow]); // Keep cache in sync if (_cachedRows) rows[idx] = newRow; return { success: true, type: "updated", version: oldVersion + 1, changedFields: changed }; } const rowId = Utilities.getUuid(); const newRow = [ now, data.platformSKU, data.parentSKU, childStr, data.gst, data.ean, data.mrp, data.platformName, data.channelSKU, data.shelfLife, data.status, data.uploaderName, wh, "Pending", rowId, 1, barcode, gpc ]; mainSheet.appendRow(newRow); // Keep cache in sync for later rows in the same batch if (_cachedRows) _cachedRows.push(newRow); return { success: true, type: "created", version: 1 }; } /* ── bulk submit — reads sheet ONCE, processes all rows, writes in one pass ── */ function bulkSubmit(body) { const mainSheet = getSheet(MAIN_SHEET); const histSheet = getSheet(HIST_SHEET); // Build an in-memory index: platformSKU → rowIndex (1-based data index) const allRows = mainSheet.getDataRange().getValues(); const skuIndex = {}; for (let i = 1; i < allRows.length; i++) { skuIndex[String(allRows[i][1]).trim()] = i; } const results = []; for (const row of (body.rows || [])) { try { const key = String(row.platformSKU).trim(); const idx = skuIndex.hasOwnProperty(key) ? skuIndex[key] : -1; const r = upsertSKU(row, mainSheet, histSheet, allRows, idx); // If newly created, its row is now at the end — update index for potential duplicates in same batch if (r.type === "created") { skuIndex[key] = allRows.length - 1; } results.push({ platformSKU: row.platformSKU, ...r }); } catch(e) { results.push({ platformSKU: row.platformSKU, success: false, error: e.toString() }); } } try { const logSheet = getSheet(BULKLOG_SHEET); if (logSheet.getLastRow() === 0) logSheet.appendRow(["Timestamp","Submitted By","Total","Created","Updated","Errors"]); logSheet.appendRow([ new Date().toISOString(), body.submittedBy || "", results.length, results.filter(r => r.type === "created").length, results.filter(r => r.type === "updated").length, results.filter(r => !r.success).length ]); } catch(e) {} return { success: true, results }; } /* ── fetch listings ── */ function fetchSKUs() { const rows = getSheet(MAIN_SHEET).getDataRange().getValues(); return { success: true, data: rows.slice(1).map(rowToObj) }; } /* ── fetch history ── */ function fetchHistory() { const rows = getSheet(HIST_SHEET).getDataRange().getValues(); return { success: true, data: rows.slice(1).map(r => ({ archivedAt: r[0], platformSKU: r[1], parentSKU: r[2], childCodes: r[3], gst: r[4], ean: r[5], mrp: r[6], platformName: r[7], channelSKU: r[8], shelfLife: r[9], status: r[10], uploaderName: r[11], warehouseMapping: r[12], approval: r[13], rowId: r[14], version: r[15] || 1, originalTimestamp: r[16], changedFields: r[17], barcode: r[18] || "", grandParentCode: r[19] || "" })) }; }
3
Deploy: Deploy → New Deployment → Type: Web App → Execute as: Me → Access: Anyone (⚠ must be "Anyone", NOT "Anyone in your org" — otherwise browser requests fail). Copy the URL. Then open the deployment URL directly in your browser once to authorize it.
4
Paste URL: Open this HTML file in a text editor, find const SCRIPT_URL = "" and paste your URL. Save and reopen.
5
Managing uploaders: Simply add or remove names in the Uploaders sheet tab (column A, one name per row, starting from A2). Changes reflect in the portal on next page load — no code changes needed.
6
Hosting on GitHub Pages (free, replaces Netlify):
1. Create a new GitHub repository (public or private with Pages enabled).
2. Name the file index.html and push it to the repo's main branch.
3. Go to Settings → Pages, set Source to Deploy from a branch, branch: main, folder: / (root). Click Save.
4. GitHub will publish your site at https://<username>.github.io/<repo-name>/ within ~60 seconds.
5. To update the portal later: edit index.html locally, commit, and push — GitHub Pages redeploys automatically.

Tip: If you want a custom domain, add a CNAME file to the repo root with your domain name and configure your DNS provider to point to GitHub Pages.
